import express from "express"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); /* 🔐 API KEY */ const API_KEY = "RAMY_SECRET_2026"; /* 🗄️ DATABASE (يمكن تطويرها لاحقًا MongoDB) */ let users = [ { id: "1001", username: "ramy", name: "Ramy Developer", avatar: "https://xatimg.com/image/NTKWfOtODi9Z.jpg", status: "online", country: "EG" }, { id: "1002", username: "ahmed", name: "Ahmed Ali", avatar: "https://xatimg.com/image/NTKWfOtODi9Z.jpg", status: "offline", country: "EG" }, { id: "1003", username: "sara", name: "Sara Mohamed", avatar: "https://xatimg.com/image/NTKWfOtODi9Z.jpg", status: "online", country: "SA" } ]; /* 🔐 AUTH MIDDLEWARE */ function auth(req, res, next){ const key = req.query.key; if(!key || key !== API_KEY){ return res.status(401).json({ success: false, message: "Invalid or missing API Key" }); } next(); } /* 🏠 ROOT */ app.get("/", (req, res) => { res.json({ system: "RAMY API SYSTEM", status: "ACTIVE", endpoints: { search: "/api/search?q=&key=", user: "/api/user/:id?key=", users: "/api/users?key=" } }); }); /* 🔍 SEARCH (ID + USERNAME + NAME) */ app.get("/api/search", auth, (req, res) => { const q = (req.query.q || "").toLowerCase(); const result = users.filter(u => u.id.includes(q) || u.username.toLowerCase().includes(q) || u.name.toLowerCase().includes(q) ); res.json({ success: true, query: q, count: result.length, data: result }); }); /* 👤 GET USER BY ID */ app.get("/api/user/:id", auth, (req, res) => { const user = users.find(u => u.id === req.params.id); if(!user){ return res.status(404).json({ success: false, message: "User not found" }); } res.json({ success: true, data: user }); }); /* 📦 GET ALL USERS */ app.get("/api/users", auth, (req, res) => { res.json({ success: true, count: users.length, data: users }); }); /* ➕ ADD USER */ app.post("/api/add", auth, (req, res) => { const { id, username, name, avatar, status, country } = req.body; if(!id || !username){ return res.status(400).json({ success: false, message: "id and username required" }); } users.push({ id, username, name: name || "Unknown", avatar: avatar || "", status: status || "offline", country: country || "N/A" }); res.json({ success: true, message: "User added successfully", data: users[users.length - 1] }); }); /* ❌ NOT FOUND */ app.use((req, res) => { res.status(404).json({ success: false, message: "Route not found" }); }); /* 🚀 START SERVER */ const PORT = 5000; app.listen(PORT, () => { console.log(`🚀 RAMY API RUNNING ON http://localhost:${PORT}`); });
